  function fib(n)
     local a, b = 0, 1
     while n > 0 do
         a, b = b, a + b
         n = n - 1
     end

     return a
 end

 function CoFunc()
     print('Coroutine started')
     local i = 0
     for i = 0, 10, 1 do
         print(fib(i))                    
         coroutine.wait(1)
     end
     print('Coroutine ended')
 end

 function myFunc()
     coroutine.start(CoFunc)
 end



function waitSeconds(t)
	local timeStamp=Time.time+t
	while Time.time<timeStamp do
		coroutine.yield()
	end
end

function MyFunction()
	local i=0
	for	i=0,10,1 do

	end
end
